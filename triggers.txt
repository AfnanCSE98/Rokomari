CREATE OR REPLACE TRIGGER COMMENT_INSERT
BEFORE INSERT
ON "COMMENT"
FOR EACH ROW
DECLARE
	T_ID NUMBER;
BEGIN
	SELECT NVL(MAX(TO_NUMBER(ID)), 0) INTO T_ID
	FROM "COMMENT";
	:NEW.ID := TO_CHAR(T_ID + 1) ;
END;
